    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getFirestore, collection, addDoc, query, onSnapshot, serverTimestamp, doc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'mat-j-app';
        const apiKey = ""; 

        let chatUnsubscribe = null;

        // --- Gemini API Integration ---
        window.askGeminiInstructor = async (prompt) => {
            const statusEl = document.getElementById('ai-status');
            if (statusEl) statusEl.innerText = "✨銀河教官が思考中...";
            
            try {
                let responseText = "";
                let retryCount = 0;
                const maxRetries = 5;
                const backoff = (retry) => Math.pow(2, retry) * 1000;

                const callApi = async () => {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: prompt }] }],
                            systemInstruction: { parts: [{ text: "あなたは宇宙時代のアクロバット飛行チーム『MAT-J』の専属AI教官です。航空力学、編隊飛行、宇宙航行、MSFSの操作に詳しく、近未来的で専門的なアドバイスを日本語で行ってください。語尾には時々『…オーバー』など通信風の演出を入れても構いません。" }] }
                        })
                    });
                    if (!response.ok) throw new Error('API Error');
                    const result = await response.json();
                    return result.candidates?.[0]?.content?.parts?.[0]?.text;
                };

                while (retryCount < maxRetries) {
                    try {
                        responseText = await callApi();
                        break;
                    } catch (e) {
                        retryCount++;
                        if (retryCount === maxRetries) throw e;
                        await new Promise(r => setTimeout(r, backoff(retryCount)));
                    }
                }

                const messagesCol = collection(db, 'artifacts', appId, 'public', 'data', 'messages');
                await addDoc(messagesCol, {
                    text: `✨AI教官: ${responseText}`,
                    user: "MISSION CONTROL (AI)",
                    timestamp: serverTimestamp()
                });
                if (statusEl) statusEl.innerText = "";
            } catch (error) {
                console.error("Gemini Error:", error);
                if (statusEl) statusEl.innerText = "✨通信障害。リトライしてください。";
            }
        };

        window.analyzeWeather = async () => {
            const weatherPrompt = "銀河系セクターJ-01（日本上空）の気象をスキャン。6機編隊の機動に影響する空間歪み（風）や視程、雲の状況をシミュレーションし、最適な機動演目を提案してください。";
            await window.askGeminiInstructor(weatherPrompt);
        };

        // --- Firebase Core Logic ---
        window.sendMessage = async () => {
            const input = document.getElementById('chat-input');
            const nameInput = document.getElementById('user-name');
            const text = input.value.trim();
            const userName = nameInput.value.trim() || "匿名パイロット";
            
            if (!auth.currentUser) return;

            if (text) {
                try {
                    const messagesCol = collection(db, 'artifacts', appId, 'public', 'data', 'messages');
                    await addDoc(messagesCol, {
                        text: text,
                        user: userName,
                        timestamp: serverTimestamp()
                    });
                    input.value = '';
                    if (text.includes("教官") || text.includes("？") || text.includes("?")) {
                        await window.askGeminiInstructor(text);
                    }
                } catch (e) {
                    console.error("Error adding document: ", e);
                }
            }
        };

        const initApp = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (err) {
                console.error("Auth failed:", err);
            }
        };

        onAuthStateChanged(auth, (user) => {
            if (user) startChatListener();
        });

        function startChatListener() {
            const messagesCol = collection(db, 'artifacts', appId, 'public', 'data', 'messages');
            if (chatUnsubscribe) chatUnsubscribe();
            chatUnsubscribe = onSnapshot(messagesCol, (snapshot) => {
                const chatContainer = document.getElementById('chat-messages');
                if (!chatContainer) return;
                chatContainer.innerHTML = '';
                const docs = snapshot.docs
                    .map(doc => ({ id: doc.id, ...doc.data() }))
                    .sort((a, b) => (a.timestamp?.seconds || 0) - (b.timestamp?.seconds || 0));
                docs.forEach(data => {
                    const isAI = data.user === "MISSION CONTROL (AI)";
                    const div = document.createElement('div');
                    div.className = `mb-2 p-3 rounded-xl shadow-[0_0_15px_rgba(14,165,233,0.1)] border-l-4 ${isAI ? 'bg-slate-800/80 border-cyan-400 text-cyan-50' : 'bg-slate-900/60 border-sky-600 text-white'} self-start max-w-[90%] backdrop-blur-sm`;
                    div.innerHTML = `<span class="font-bold text-[10px] ${isAI ? 'text-cyan-300' : 'text-sky-400'} block tracking-widest">${data.user}</span><p class="text-sm break-words">${data.text}</p>`;
                    chatContainer.appendChild(div);
                });
                chatContainer.scrollTop = chatContainer.scrollHeight;
            });
        }

        initApp();
    </script>

                const element = document.getElementById(section);
                if (element && window.pageYOffset >= element.offsetTop - 100) {
                    current = section;
                }
            });

            navItems.forEach(item => {
                item.classList.remove('active');
                if (item.getAttribute('href').includes(current)) {
                    item.classList.add('active');
                }
            });
        });

        document.getElementById('chat-input').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') window.sendMessage();
        });
    </script>
